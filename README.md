    private void jButton19ActionPerformed(java.awt.event.ActionEvent evt) {                                          
        num2=Double.parseDouble(jTextField_display.getText());
        switch(operator)
        {
            case    "+":
                res=num1+num2;
                    ans=String.format("%.0f",res);
                    jTextField_display.setText(ans);
                    break;
            case    "-":
                res=num1-num2;
                    ans=String.format("%.0f",res);
                    jTextField_display.setText(ans);
                    break;
            case    "*":
                res=num1*num2;
                    ans=String.format("%.0f",res);
                    jTextField_display.setText(ans);
                    break; 
            case    "/":
                res=num1/num2;
                    ans=String.format("%.0f",res);
                    jTextField_display.setText(ans);
                    break;
        }
    }
